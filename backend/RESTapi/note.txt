how to setup nodejs with typescript
1. mkdir check
2. cd check
3. npm init -y;
4. npm i dotenv express mongoose
5. npm i typescript ts-node @types/node @types/express @types/mongoose --save dev 
6. tsc -init
7. tsconfig.json
outdir ./dist
rootdir ./src 
moudle es2020
include src/**/*
exudle nodemoudles

8. package.json
"scripts": {
  "start": "node dist/index.js",
  "build": "tsc",
  "dev": "ts-node src/index.ts"
}
